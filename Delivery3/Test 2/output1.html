public class Prime {

    public static void main(String[] args) {//block 1

        int low = 20, high = 50;


        while (low < high) {//block 2
            if(checkPrimeNumber(low))

                {//block 3
		System.out.print(low + " ");}


            ++low;
        }

    }



    public static boolean checkPrimeNumber(int num) {//block 4
        boolean flag = true;


        for(int i = 2; i <= num/2; ++i) {//block 5


            if(num % i == 0) {//block 6
                flag = false;
                break;
            }

        }


        return flag;
    }

}
